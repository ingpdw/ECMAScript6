ECMAScrpt6란?

  • JavaScript 언어의 새로운 에디션(2015년 6월)
  • 2009년 ECMAScript5 발표 후 새로운 에디션을 발표
  • 다른 language에서 검증된 새로운 요소들을 추가
  • 하위 호환됨(ES5)

ES6의 새로운 요소 - rest arguments


//ES6
function sum( a, b, ...nums ){
	let sumAll = a + b;
	for( let i = 0, len = nums.length; i < len; i++){
		sumAll += nums[ i ];
	};

	return sumAll;
}

sum( 5, 10, 15, 20, 25 ); //75

function _classCallCheck(instance, Constructor) { if (!(instance instanceof Constructor)) { throw new TypeError('Cannot call a class as a function'); } }

var Polygon = (function () {
  function Polygon(height, width) {
    _classCallCheck(this, Polygon);

    this.height = height;
    this.width = width;

      return this.height * this.width;

gulp + browserify 설정


//gulpfile.js

'use strict';

var gulp = require('gulp');
var browserify = require('browserify');
var babelify = require('babelify');
var source = require('vinyl-source-stream');
var buffer = require('vinyl-buffer');
var uglify = require('gulp-uglify');

gulp.task('build', function () {
  browserify({
    entries: './src/js/class.js',
    debug: false
  })
  .transform( babelify )
  .bundle()
  .pipe(source( 'app.js' ))
  .pipe(buffer())
  .pipe(uglify())
  .pipe(gulp.dest('./src/dist/'))
});

gulp.task('watch', ['build'], function() {
  gulp.watch(['./src/js/*.js' ],['build'] );
  gulp.watch(['./views/**/*.ejs' ], ['build'] );
});

gulp.task('default', ['watch']);

gulp + webpack 설정


//gulpfile.js

'use strict';

var gulp = require('gulp');
var gutil = require('gulp-util');
var webpack = require('webpack');

gulp.task('webpack', function() {

    webpack({
      context: __dirname + '/src',
      entry: {
        javascript: './js/class.js'
      },
      output: {
        filename: 'app.js',
        path: __dirname + '/src/dist',
      },
      plugins: [new webpack.optimize.UglifyJsPlugin()],
      module: {
        loaders: [{test: /\.js/, loaders: ['babel-loader']}]
      }
    }, function(err, stats) {
        if(err) throw new gutil.PluginError('webpack', err);
    });
});
gulp.task('default', ['webpack']);
